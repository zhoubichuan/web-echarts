## 6.柱状图

::: demo

```vue
<template>
  <div ref="pie" style="height: 400px"></div>
</template>

<script>
export default {
  data() {
    return {
      option: {
        title: {
          text: "上下双柱状图",
          left: "center",
        },
        tooltip: {
          trigger: "axis",
          axisPointer: {
            animation: false,
          },
        },
        legend: {
          data: ["A", "B"],
          left: 10,
        },
        axisPointer: {
          link: [
            {
              xAxisIndex: "all",
            },
          ],
        },
        grid: [
          {
            left: 60,
            right: 50,
            height: "35%",
          },
          {
            left: 60,
            right: 50,
            top: "55%",
            height: "35%",
          },
        ],
        xAxis: [
          {
            type: "category",
            axisLine: { onZero: false },
            data: ["3月1日", "3月2日", "3月3日", "3月4日"],
          },
          {
            gridIndex: 1,
            type: "category",
            axisLine: { onZero: true },
            data: ["4月1日", "4月2日", "4月3日", "4月4日"],
            position: "top",
          },
        ],
        yAxis: [
          {
            name: "A(m³/s)",
            type: "value",
          },
          {
            gridIndex: 1,
            name: "B",
            type: "value",
            inverse: true,
          },
        ],
        series: [
          {
            name: "A",
            type: "bar",
            symbolSize: 8,
            data: [1, 5, 4, 7],
          },
          {
            name: "B",
            type: "bar",
            xAxisIndex: 1,
            yAxisIndex: 1,
            symbolSize: 8,
            data: [2, 3, 5, 7],
          },
        ],
      },
    };
  },
  mounted() {
    let charts = this.$echarts.init(this.$refs.pie);
    charts.setOption(this.option);
  },
};
</script>
```
