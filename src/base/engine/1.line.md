# 折线图

## 1.双坐标轴

::: demo

```vue
<template>
  <WebBar :config="getOptions" :data="data"></WebBar>
</template>

<script>
export default {
  data() {
    return {
      data: [],
    };
  },
  async created() {
    let res = await this.$api.getLine1();
    if (res.data) {
      this.data = res.data;
    }
  },
  methods: {
    getOptions(data) {
      return {
        title: {
          text: "时间",
          subtext: "单位：时",
        },
        tooltip: {
          trigger: "axis",
          axisPointer: {
            animation: false,
          },
        },
        axisPointer: {
          link: [
            {
              xAxisIndex: "all",
            },
          ],
        },
        grid: [
          {
            left: 60,
            right: 50,
            height: "35%",
          },
          {
            left: 60,
            right: 50,
            top: "55%",
            height: "35%",
          },
        ],
        xAxis: [
          {
            type: "category",
            boundaryGap: false,
            axisLine: { show: false },
            data: data,
          },
          {
            gridIndex: 1,
            type: "category",
            boundaryGap: false,
            axisLine: { show: false },
            position: "top",
          },
        ],
        yAxis: [
          {
            type: "value",
          },
          {
            gridIndex: 1,
            type: "value",
            inverse: true,
          },
        ],
        series: [
          {
            name: "a",
            type: "line",
            symbolSize: 8,
            data: [1.97, 2.96, 4.96, 6.95],
          },
          {
            name: "b",
            type: "line",
            xAxisIndex: 1,
            yAxisIndex: 1,
            symbolSize: 8,
            data: [0.6, 0.9, 1.4, 4.5, 0.5, 0.3],
          },
        ],
      };
    },
  },
};
</script>
```
