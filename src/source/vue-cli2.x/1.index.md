# zrender 基础

::: demo

```vue
<template>
  <div style="height:100%;" ref="container"></div>
</template>

<script>
export default {
  data() {
    return {}
  },
  mounted() {
    var zr = this.$zrender.init(this.$refs.container)
    var circle = new this.$zrender.Circle({
      shape: {
        cx: 150, // 圆心x坐标
        cy: 50, // 圆心y坐标
        r: 40, // 圆半径
      },
      style: {
        fill: "none", // 是否填充
        stroke: "#000", // 线条颜色
      },
    })
    zr.add(circle)
  },
}
</script>
```
